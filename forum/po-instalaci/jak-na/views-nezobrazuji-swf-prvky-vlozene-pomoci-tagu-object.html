  • co ten modul Theme developer? (ten po dokonceni webu samozrejme vypnout)

$swf = new SWFObject($path);
$swf->param('type','movie')
    ->flashVar('maintainAspectRation', 'true')
    ->flashVar('volume', 50)
    ->noFlash('Install flash.')
    ->minimumVersion('9.0.28');
return theme('swfembed_embed', $swf);
