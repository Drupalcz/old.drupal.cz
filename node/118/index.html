<?php // $Id: texy.module,v 1.3.2.2 2006/09/22 19:43:45 havran Exp $  /**  * @file  * DRUPAL MODULE - Texy!  */  /**  * Implementation of hook_help().  */ function texy_help($section) {   switch ($section) {     case 'admin/help#texy':       $output = '<p>Texy! is text-to-XHTML formatter and converter library. It allows you to write structured documents       without knowledge or using of HTML language.</p><p>You write documents in humane easy-to-read plain text format       and Texy! converts it to structurally and valid XHTML code.</p><p>Texy! is one of the most complex formatters.       Its possibilities covers images, links (anchors), nested lists, tables and has full support for CSS.</p><p>Usage examples:</p><ul><li><a       href="http://texy.info/en">http://texy.info/en</a></li></ul>';       return $output;     case 'admin/modules#description':       return 'Texy! module - text-to-XHTML formatter and converter library';   } }  /**  * Implementation of hook_settings().  */ function texy_settings() {   $form = array();    $module_path = drupal_get_path('module', 'texy');    $form['general'] = array(     '#type' => 'fieldset',     '#title' => t('Syntax highlighting'),     '#collapsible' => TRUE,     '#collapsed' => FALSE,   );   $form['general']['texy_use_syntaxhighlight'] = array(     '#type' => 'checkbox',     '#title' => t('Use code syntax highlighting'),     '#description' => t('Check if you want highlight code.'),     '#default_value' => variable_get('texy_use_syntaxhighlight', false),   );   $form['general']['texy_syntaxhighlighter_name'] = array(     '#type' => 'textfield',     '#size' => '12',     '#title' => t('Syntax highlighter name'),     '#description' => t('Syntax highlighter which Texy! use for syntax       highlighting. Syntax highlighter must be installed in to <code>'.$module_path.'/lib/syntaxhighlightername</code>       directory and there must be <code>'.$module_path.'/lib/syntaxhighlightername.php</code> file,       with php code for Texy! implementation this highlighter. For example I use for syntax highlighting FSHL .       I have directory <code>'.$module_path.'/lib/fshl</code> and code <code>'.$module_path.'/lib/fshl.php</code>       and name for syntax highlighter is set to <em>fshl</em>.'),     '#default_value' => variable_get('texy_syntaxhighlighter_name', ''),   );   $form['formatset'] = array(     '#type' => 'fieldset', '#title' => t('Basic filtering settings'), '#collapsible' => TRUE, '#collapsed' => FALSE, );   $form['formatset']['texy_image_dir'] = array(     '#type' => 'textfield',     '#size' => '12',     '#title' => t('Images directory'),     '#description' => t('Directory where to look for saved images. This is usually something like <tt>'.file_directory_path().'/images</tt>'),     '#default_value' => variable_get('texy_image_dir', file_directory_path().'/images'),   );   $form['formatset']['texy_allowed_classes'] = array(     '#type' => 'textfield',     '#title' => t('Allowed classes'),     '#description' => t('Allow following CSS classes and IDs to appear in the text. Default value is empty - allow no classes.     Specify <i>&lt;all&gt;</i> to allow all classes - this is not recommended. Specify allowed classes separated by commas.'),     '#default_value' => variable_get('texy_allowed_classes', ''),   );   $form['formatset']['texy_allowed_styles'] = array(     '#type' => 'textfield',     '#title' => t('Allowed inline styles'),     '#description' => t('Allow following CSS inline styles such as \'font-size\', \'color\', to appear in the text.      Default value is empty - allow no styles. Specify <i>&lt;all&gt;</i> to allow all styles - this is not recommended. Specify allowed styles      separated by commas.'),     '#default_value' => variable_get('texy_allowed_styles', base_path()),   );   $form['formatset']['texy_obfuscate_email'] = array(     '#type' => 'checkbox',     '#title' => t('Obfuscate emails'),     '#description' => t('Obfuscate emails from SPAMbots?'),     '#default_value' => variable_get('texy_obfuscate_email', true),   );   $form['formatset']['texy_heading_base'] = array(     '#type' => 'select',     '#title' => t('Start headings from'),     '#options' => array(1 => '<h1>', 2 => '<h2>', 3 => '<h3>', 4 => '<h4>', 5 => '<h5>', 6 => '<h6>'),     '#description' => t('Highest allowed heading. Default is &lt;h2&gt;. Not recommended to change for Drupal.'),     '#default_value' => variable_get('texy_heading_base', 2),   );      return $form; }  /**  * Implementation of hook_menu().  */ function texy_menu($may_cache) {   $items = array();    if ($may_cache) {   }   else {     // trick for add styles     // need better way     theme_add_style(drupal_get_path('module', 'texy') . '/lib/fshl/styles/COHEN_style.css');   }    return $items; }  /**  * Implementation of hook_filter().  */ function texy_filter($op, $delta = 0, $format = -1, $text = '') {   switch ($op) {     case 'list':       return array(0 => 'Texy! text-to-XHTML formatter and converter filter');      case 'description':       switch ($delta) {         case 0:           return 'Texy! is text-to-XHTML formatter and converter library. It allows you to write structured                   documents without knowledge or using of HTML language.';         default:           return;       }       break;      case 'prepare':       switch ($delta) {         case 0:           return $text;         default:           return $text;         }       break;      case 'process':       switch ($delta) {         case 0:           return _filter_texy($text);         default:           return $text;       }       break;      case 'settings':       switch ($delta) {         default:           return;       }       break;      default:       return $text;   } }  /**  * Implementation of hook_filter_tips().  */ function texy_filter_tips($delta, $format, $long = false) {   return 'Texy! tips: <a href="http://texy.info/en" target="_top">http://texy.info/en</a>...'; }  /**  * Main Texy! filter.  */ function _filter_texy($text) {   // get base module path   $module_path = drupal_get_path('module', 'texy');    // include Texy!   require_once $module_path . '/texy/texy.php';   $texy = &new Texy();    // check for use syntax highlighter   if (variable_get('texy_use_syntaxhighlight', false)) {     $texy_syntaxhighlighter_name = variable_get('texy_syntaxhighlighter_name', '');     // check name of syntax highlighter     if ($texy_syntaxhighlighter_name == '') {       // set user callback function for /-- code blocks (PHP only)       $texy->blockModule->codeHandler = '_filter_texy_basic_highlighter';     }     else {       $texy_syntaxhighlighter = $module_path . '/lib/' . $texy_syntaxhighlighter_name . '.php';       $texy_syntaxhighlighter_dir = $module_path . '/lib/' . $texy_syntaxhighlighter_name;       if (file_exists($texy_syntaxhighlighter) && file_exists($texy_syntaxhighlighter_dir)) {         // set user callback function for /-- code blocks (for configured syntax highlighter)         $texy->blockModule->codeHandler = '_filter_texy_' . $texy_syntaxhighlighter_name . '_highlighter';         require_once $texy_syntaxhighlighter;       }       else {         drupal_set_message(t('Syntax highlighter <em>' . $texy_syntaxhighlighter_name . '</em> is not ready...'),'error');       }     }   }    // other optional configuration   $texy->utf = TRUE;                    // enable UTF-8, always on for DRUPAL   // basic settings   $_allowedclasses = variable_get('texy_allowed_classes', '');   if ($_allowedclasses == '<all>') {     // permit all classes     $texy->allowedClasses = true;   } else if (empty($_allowedclasses)) {     // permit no classes     $texy->allowedClasses = false;   } else {     // permit only specified classes     $texy->allowedClasses = preg_split('/,/', str_replace(' ', '', $_allowedclasses));   }    $_allowedstyles = variable_get('texy_allowed_styles', '');   if ($_allowedstyles == '<all>') {     // permit all styles     $texy->allowedStyles = true;   } else if (empty($_allowedstyles)) {     // permit no styles     $texy->allowedStyles = false;   } else {     // permit only specified styles     $texy->allowedStyles = preg_split('/,/', str_replace(' ', '', $_allowedstyles));   }    $texy->imageModule->root = variable_get('texy_image_dir', 'files/images');   $texy->imageModule->rootPrefix;   $texy->headingModule->top = variable_get('texy_heading_base', 2); // start hedings from h2    // main Texy! filter processing   $html = $texy->process($text);   return $html; }  /**  * basic syntax highlighter (if there is no syntax highlighter name defined)  */ function _filter_texy_basic_highlighter(&$element) {   $lang = strtoupper($element->lang);   // only for /---code php   if ($lang != 'PHP') {     return;   }    $element->setContent(highlight_string("<?php\n" . decode_entities($element->content) . "\n?>", TRUE), TRUE); }
