  • ja by som to mal najradsej zoznam bloggerov zoradeny pola poctu blogov, neviem to v tom docielit
  • odkaz na dalsiu stranku so zoznamom blogov toho ktoreho usera mi osobne pride zbytocny a duplicitny, uplne by malo stacit odkazat na site/blog/username

p.s. je dobre si doplnit do filtrov len pre aktivnych userov, aby ak mas nejakych zmazanych, tak si na zaciatku nemal len nejake riadky len cisla.

v kazdom pripade este raz diky, ak by ste niekto mali napady ako tych userov zoradit podla poctu blogov tak by to bolo fajn. Zrejme by to slo v theme a v nejakom foreachi, ale to uz mi pride jednoduchsie vlastny blok s vlastnym query.

<?php // $Id seznam_blogeru.module,v 1.0 $  /**  * Implementation of hook_menu().  */ function seznam_blogeru_menu() {   $items = array();   $items['seznam_blogeru'] = array(     'title' => 'Seznam blogerů',     'page callback' => 'seznam_blogeru_page',     'access arguments' => array('access content'),     'type' => MENU_NORMAL_ITEM,   );   return $items; }  function seznam_blogeru_page() {   $users = db_query("SELECT n.uid, COUNT(n.uid) AS count FROM {node} n INNER JOIN {users} AS u ON n.uid = u.uid WHERE n.type = 'blog/index.html' AND u.status <> 0 AND n.status <> 0 GROUP BY n.uid ORDER BY count DESC");    $rows = array();   while ($u = db_fetch_object($users)) {     $account = user_load(array('uid' => $u->uid));     $rows[] = array(       theme('username', $account),       l($u->count, "blog/$u->uid", array('attributes' => array('title' => 'Zobrazit seznam blogů'))),     );   }   $header = array('Uživatel', 'Počet blogů');   $output = theme('table' , $header , $rows);    return $output; }
